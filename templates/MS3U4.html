#logit plot
def logitplot(data, predictor, outcome, numIntervals=10, spline=True, spar=2):
    #Make a copy of the original dataframe so it won't get changed.
    data = data.copy()

    #Check inputs
    if type(predictor) != str:
        print("The predcitor must be a string!")
        return
    if type(outcome) != str:
        print("The outcome must be a string")
        return
    
    #Initialize vectors
    logits = np.zeros(numIntervals)
    means = np.zeros(numIntervals)
    
    #Define outcome and predictor columns (renames them to make the code below simpler)
    data.rename(columns = {outcome: 'outcome', predictor: 'predictor'}, inplace = True)

    #Check for missing data
    if data['predictor'].isnull().values.any():
        print("The predictor contains missing data.")
        return
    if data['outcome'].isnull().values.any():
        print("The outcome contains missing data.")
        return
    
    #Sort data in ascending order of predictor
    data.sort_values(by=['predictor','ID'], inplace = True)
    data.reset_index(drop=True, inplace=True)

    #Define the intervals
    intervalLength = math.floor(data['predictor'].count() / numIntervals)
    #math.floor() enforces rounding down, in case numIntervals doesn't divide evenly into the number of rows.
    if data['predictor'].count() / numIntervals - math.floor(data['predictor'].count() / numIntervals) != 0: 
        print("The number of intervals does not divide evenly into the size of the dataset. Some data points will be omitted.")
  
    #Define the starting index of each section
    intervalStarts = np.zeros(numIntervals, dtype=int)
    for k in range(0, numIntervals):
        intervalStarts[k] = 1 + k*intervalLength
    
    #Loop over each section
    for j in intervalStarts: 
        positive=0
        negative=0
        summ=0
    
        #Loop over each data point in the section
        for i in range(j-1, intervalLength+j-1):
            
            if data['outcome'][i] == 1: #outcome variable is 1
                positive = positive + 1
            elif data['outcome'][i] == 0:  #outcome variable is 0
                negative = negative + 1
            else:
                print("The outcome column must be binary!")
                return
            summ = summ + data['predictor'][i] #adding the predictor values so we can calculate the mean
       
        
        x = positive/(positive+negative)        #calculates the probability
        if x == 1:
            logits[round(j//intervalLength)] = math.inf
        else:
            logits[round(j//intervalLength)] = math.log(x/(1-x))        #puts the logit in the index of j
        means[round(j//intervalLength)] = summ/intervalLength   #puts the mean in the index of j 
    
    
    #Check for infinite logits
    logits_bool = list(map(lambda x: np.isinf(x), logits))
    if any(logits_bool):
        print("Infinite logits generated. Not all points will be plotted.")
  
    #Put logits and means in a dataframe
    ma = np.transpose(np.array([logits, means]))
    df = pd.DataFrame(data=ma, index=None, columns=["logits", "means"])
    
    #Plot each logit at the corresponding predictor value
    #Spline is false
    if spline==False:
        fig= plt.figure(figsize=(9.5,4))
        plt.plot(means, logits, 'o', color='red')
    else:
        #Plot a smoothing spline
        df.replace([np.inf, -np.inf], np.nan, inplace=True)
        df.dropna(subset=['logits'], inplace=True)
        sns.lmplot(x='means',y='logits',data=df, ci=None, order=spar, height=4, aspect=2)
        #sns.lmplot(x='means',y='logits',data=df, ci=None, lowess=True, height=4, aspect=2)
