#General linear model with interaction term
classdata = pd.read_excel('classdata.xlsx')
classdata.dropna(subset=['IsBookSmart', 'politics', 'obama'], inplace=True)
Y = classdata['obama']
X1 = classdata['IsBookSmart']
X2 = classdata['politics']
#Interaction term
classdata['BSPol'] = classdata['IsBookSmart']*classdata['politics']
I = classdata['BSPol']
reg = LinearRegression()
reg.fit(list(zip(X1,X2, I)), Y)
isbooksmart, politics, interaction = reg.coef_[0], reg.coef_[1], reg.coef_[2]
intercept = reg.intercept_
print(f'intercept is {intercept:.{5}}, isbooksmart coef is {isbooksmart:.{5}}, politics coef is {politics:.{5}}, interaction is {interaction:.{4}}')

intercept is 7.2706, isbooksmart coef is 26.509, politics coef is 0.99619, interaction is -0.4042

#Q14 Fit the following linear regression model:
#Expected optimism = intercept + alcohol (drinks/week) + Varsity sports (1=played varsity sports in high school, 
#0=did not) + Varsity*alcohol
#What is the resulting beta coefficient for the interaction term?
classdata = pd.read_excel('classdata.xlsx')
classdata.dropna(subset=['varsity', 'alcohol', 'optimism'], inplace=True)
Y = classdata['optimism']
X1 = classdata['varsity']
X2 = classdata['alcohol']
#Interaction term
classdata['VarsAlc'] = classdata['varsity']*classdata['alcohol']
I = classdata['VarsAlc']
reg = LinearRegression()
reg.fit(list(zip(X1,X2, I)), Y)
varsity, alcohol, interaction = reg.coef_[0], reg.coef_[1], reg.coef_[2]
intercept = reg.intercept_
print(f'intercept is {intercept:.{5}}, varsity coef is {varsity:.{5}}, alcohol coef is {alcohol:.{5}}, interaction is {interaction:.{4}}')
