#Confidence Interval for correlation coefficient, calculating by hand
N = classdata['bushjr'].count()
#Convert r to z
z_r = math.log((1 + r) / (1 - r)) / 2.0
#Z's confidence interval. The sampling distribution of z' is approximately 
#normally distributed and has standard error of 1/sqrt(n-3).
se = 1.0 / math.sqrt(N - 3)
alpha = 0.05
z_crit = stats.norm.ppf(1 - alpha/2)  # 2-tailed z critical value
lo = z_r - z_crit * se
hi = z_r + z_crit * se
#Convert the confidence interval back to r.
e_lo = math.exp(2 * lo)
e_hi = math.exp(2 * hi)
lo_r = ((e_lo - 1) / (e_lo + 1))
hi_r = ((e_hi - 1) / (e_hi + 1))
print(f'Te confidence interval is ({lo_r}, {hi_r})')

Te confidence interval is (-0.6922963383120526, -0.26176646761920425)
